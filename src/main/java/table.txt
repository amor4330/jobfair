CREATE TABLE User (
	user_id	varchar(30)	NOT NULL,
	user_pw	varchar(30)	NULL	DEFAULT 1,
	mg_auth	varchar(1)	NOT NULL,
	user_name	varchar(10)	NULL,
	user_rrn	varchar(10)	NULL	DEFAULT 1,
	user_email	varchar(30)	NULL,
	user_phone	varchar(20)	NULL,
	user_regDate	timestamp	NULL,
	user_gender	varchar(1)	NULL,
	user_address	varchar(100)	NULL,
	com_id	varchar(30)	NOT NULL DEFAULT 0
);
